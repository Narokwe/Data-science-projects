Remove outliers in each column using Interquartile range (IQR)¶

In [11]:
# Select the columns of interest
columns = ['Pregnancies', 'Glucose', 'BloodPressure', 'SkinThickness', 'Insulin', 'BMI', 'DiabetesPedigreeFunction', 'Age', 'Outcome']
In [12]:
# Define the IQR multiplier for outlier detection
iqr_multiplier = 1.5
In [13]:
# Iterate over each column and remove outliers based on IQR
for column in columns:
    # Calculate the IQR for the column
    q1 = df[column].quantile(0.25)
    q3 = df[column].quantile(0.75)
    iqr = q3 - q1
In [14]:
# Define the lower and upper bounds for outlier detection
lower_bound = q1 - iqr_multiplier * iqr
upper_bound = q3 + iqr_multiplier * iqr
In [15]:
# Filter out the outliers in the current column
df = df[(df[column] >= lower_bound) & (df[column] <= upper_bound)]
