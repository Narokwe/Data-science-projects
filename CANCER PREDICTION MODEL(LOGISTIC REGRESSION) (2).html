In [26]:
# Combine the training features and labels
train_data = np.column_stack((X_train_scaled, y_train))
In [27]:
# Separate the majority and minority classes
majority_class = train_data[train_data[:, -1] == 0]
minority_class = train_data[train_data[:, -1] == 1]
In [28]:
# Upsample the minority class to balance the data
minority_class_upsampled = resample(minority_class, replace=True, n_samples=len(majority_class), random_state=42)
In [29]:
# Combine the upsampled minority class with the majority class
balanced_train_data = np.vstack((majority_class, minority_class_upsampled))
In [30]:
# Split the balanced training data into features and labels
X_train_balanced = balanced_train_data[:, :-1]
y_train_balanced = balanced_train_data[:, -1]
In [31]:
